随手写的小工具


import base64
import binascii
from urllib.parse import unquote

decode_str = input("input:")


def bs64(string):
    return "[*]base64:" + base64.b64decode(string.encode()).decode()


def bs32(string):
    return "[*]base32:" + base64.b32decode(string.encode()).decode()


def bs16(string):
    return "[*]base16:" + base64.b16decode(string.encode()).decode()


def hexStr_to_str(string):
    hexing = string.encode('utf-8')
    str_bin = binascii.unhexlify(hexing)
    return "[*]hex:" + str_bin.decode('utf-8')


def urldecode(string):
    compare_string = string
    sting = unquote(string)
    if compare_string == sting:
        return ""
    return "[*]url:" + sting


def decode_ctf(string):
    try:
        print(bs64(string))
        decode_ctf(bs64(string)[10:])
    except:
        pass
    try:
        print(bs32(string))
        decode_ctf(bs32(string)[10:])
    except:
        pass
    try:
        print(bs16(string))
        decode_ctf(bs16(string)[10:])
    except:
        pass
    try:
        print(hexStr_to_str(string))
        decode_ctf(hexStr_to_str(string)[7:])
    except:
        pass
    try:
        print(urldecode(string))
        if "%" in string:
            decode_ctf(urldecode(string)[7:])
    except:
        pass


decode_ctf(decode_str)
