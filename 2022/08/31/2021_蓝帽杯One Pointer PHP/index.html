            $nvpair = chr($nlen);

            $p += ($nlen + $vlen);
        }
        return $array;
    }

        $ret['requestId']     = (ord($data{2}) << 8) + ord($data{3});
        $ret['contentLength'] = (ord($data{4}) << 8) + ord($data{5});
        $ret['paddingLength'] = ord($data{6});
        $ret['reserved']      = ord($data{7});
        return $ret;
    }

            }
            return $resp;
        } else {
            return false;
        }
